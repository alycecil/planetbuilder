alicemod_decision_enable_archology = {
	owned_planets_only = yes
	icon = decision_prospect

	enactment_time = 200
	resources = {
		category = decisions
		cost = {
		    minerals = 10000
		    energy = 10000
			influence = 250
		}
	}

	potential = {
		NOR = {
			is_planet_class = pc_ringworld_habitable_damaged
			is_planet_class = pc_ringworld_tech_damaged
			is_planet_class = pc_ringworld_seam_damaged
		    has_planet_flag = flag_alicemod_city_world
		}
	}

	effect = {
		set_planet_flag = flag_alicemod_city_world
        clear_blockers = yes
        while = {
            count = 10
            add_district = alicemod_district_arcology_housing
        }
	}

	ai_weight = {
		weight = 5
		modifier = {
            factor = 0
            num_pops < 100
        }
		modifier = {
            factor = 20
            solar_system = {
                exists = owner
                is_same_value = owner.capital_scope.solar_system
            }
        }
        modifier = {
            factor = 1.5
            solar_system = {
                any_neighbor_system = {
                    exists = owner
                    is_same_value = owner.capital_scope.solar_system
                }
            }
        }
	}
}

alicemod_decision_become_archology = {
	owned_planets_only = yes
	icon = decision_prospect

	enactment_time = 1000
	resources = {
		category = decisions
		cost = {
		    minerals = 50000
		    alloys = 50000
			influence = 250
		}
	}

	potential = {
		has_planet_flag = flag_alicemod_city_world
		NOR = {
		    is_planet_class = pc_alicemod_city
			is_planet_class = pc_habitat
			is_planet_class = pc_ringworld_habitable
			is_planet_class = pc_ringworld_tech
			is_planet_class = pc_ringworld_seam
			is_planet_class = pc_relic
			is_planet_class = pc_city
		}
	}

	effect = {
		change_pc = pc_alicemod_city
        add_deposit = d_former_relic_world

        set_planet_entity = {
            entity = "city_planet_entity"
            graphical_culture = owner
        }

        change_planet_size = 10
	}

	ai_weight = {
		weight = 10
		modifier = {
            factor = 0
            num_pops < 70
        }
        modifier = {
            factor = 20
            solar_system = {
                exists = owner
                is_same_value = owner.capital_scope.solar_system
            }
        }
        modifier = {
            factor = 1.5
            solar_system = {
                any_neighbor_system = {
                    exists = owner
                    is_same_value = owner.capital_scope.solar_system
                }
            }
        }
	}
}
